Variable selection for Gaussian mixture models

# load the packages
library(sparcl)
library(clustvarsel)
library(SelvarMix)
library(VarSelLCM)
library(vscc)
library(bclust)

Human Mortality Database data

# Load and plot the Human Mortality Database data
load("hmd.RData")

labs <- colnames(X)
n <- length(labs)
matplot( t(X), type = "l", lty = 1, log = "y", axes = FALSE, xlab = "", ylab = "",
         col = adjustcolor("#252525", 0.1), lwd = 2, ylim = c(0.5e-04, 1) ) 
axis(2)
axis(1, at = 1:n, labels = labs, las = 2, cex.axis = 0.9)
mtext(expression({}[n]*m[x]), side = 2, line = 2.4, cex = 1.5)
mtext("Age intervals", side = 1, at = n/2, line = 4, cex = 1.5)

# run mclust on all the variables
Gmax <- 8
mclust.options( 
  emModelNames = c("EEE", "VEE", "EVV", "VVV") )  # for consistency with SelvarMix
mc <- Mclust(X, G = 1:Gmax)

Variable selection

# sparcl ------------------------
perm <- KMeansSparseCluster.permute(scale(X), K = mc$G, nvals = 100, nperms = 20)
fitkm <- KMeansSparseCluster(scale(X), K = mc$G, wbounds = perm$bestw)


# clustvarsel -------------------
sel <- clustvarsel(X, G = 1:Gmax, parallel = TRUE)
fitcvsl <- Mclust(X[,sel$subset], G = 1:Gmax)


# VarSelLCM ---------------------
fitlcm <- list()
for ( g in 1:Gmax ) {
  fitlcm[[g]] <- VarSelCluster(X, g,
                               crit.varsel = "MICL", 
                               nbcores = 8, 
                               initModel = 200)  # increased to avoid local optima
  print(g)
}
micl <- sapply(fitlcm, function(f) f@criteria@MICL)
best <- which.max(micl)


# vscc --------------------------
fitvscc <- vscc(scale(X), G = 1:Gmax)


# SelvarMix ---------------------
fitmix <- SelvarClustLasso(X, nbcluster = 1:Gmax, nbcores = 8)
#
# best model is for G = 3, but results are not very stable!
# to run multiple times and pick the best:
# set <- list()
# for ( r in 1:20 ) {
#   set[[r]] <- SelvarClustLasso(X, nbcluster = 1:Gmax, nbcores = 8)
# }
# bics <- sapply(set, "[[", "criterionValue")
# fitmix <- set[[ which.max(bics) ]]


# bclust ------------------------
tmp <- meancss( scale(X) )   # need to standardize the data for this method
optimfunc <- function(theta)
{
  -loglikelihood(x.mean = tmp$mean, x.css = tmp$css,
                 repno = tmp$repno, transformed.par = theta)
}
transpar <- optim( rep(0,6), optimfunc, method = "BFGS" )$par 
fitbclust <- bclust(as.matrix(scale(X)), transformed.par = transpar)

Results

# selected variables
selkm <- (fitkm[[1]]$ws != 0)*1           # sparcl
#
selcvsl <- rep(0, ncol(X))
selcvsl[sel$subset] <- 1                  # clustvarsel
#
selmicl <- fitlcm[[best]]@model@omega     # VarSelLCM
#
selvscc <- rep(0, ncol(X))                # vscc
selvscc[ match(colnames(fitvscc$bestmodel$data), colnames(X)) ] <- 1
#
selmix <- rep(0, ncol(X))                 # SelvarMix
selmix[fitmix$S] <- 1
#
selbclust <- (imp(fitbclust)$var > 0)*1   # bclust
#
SEL <- matrix( c(selkm, selcvsl, selmicl, selvscc, selmix, selbclust), 
               byrow = TRUE, ncol = ncol(X) )


# classification
ARI <- c(adjustedRandIndex(year, fitkm[[1]]$Cs),
         adjustedRandIndex(year, fitcvsl$classification),
         adjustedRandIndex(year, fitlcm[[best]]@partitions@zMAP),
         adjustedRandIndex(year, fitvscc$bestmodel$classification),
         adjustedRandIndex(year, fitmix$partition),
         adjustedRandIndex(year, fitbclust$optim.alloc)
)


# number of clusters
G <- c(mc$G, fitcvsl$G, best, fitvscc$bestmodel$G, fitmix$nbcluster, fitbclust$optim.clustno)


# plot
J <- ncol(X)
labs <- colnames(X)
met <- c("sparcl", "clustvarsel", "VarSelLCM", "vscc", "SelvarMix", "bclust")
k <- length(met)
#
par( mar = c(4,4.5,1,4) )
image( 1:J, 1:k, t(SEL), col = c("white", "#252525"), axes = FALSE, xlab = "", ylab = "" )
abline(v = 1:J + 0.5, col = "white", lwd = 2)
abline(h = 1:(k-1) + 0.5, col = "white", lwd = 4)
abline(h = c(0,k) + 0.5, col = "black", lwd = 2)
abline(v = c(0,J) + 0.5, col = "black", lwd = 2)
mtext(labs, side = 1, at = 1:J, las = 2, cex = 0.7, line = 0.5)
mtext(met, side = 2, at = 1:k, las = 1, line = 0.5, cex = 0.9)
mtext(paste0("G = ", G), side = 4, at = (1:k)+0.1, las = 1, line = 0.5, cex = 0.8)
mtext(paste0("ARI = ", round(ARI,2)), side = 4, at = (1:k)-0.3, 
      las = 1, line = 0.5, cex = 0.8)
mtext("Age intervals", side = 1, at = J/2, line = 2.5, cex = 1)

 

Variable selection for latent class analysis

# load the packages
library(LCAvarsel)
library(ClustMMDD)
library(VarSelLCM)

US Congress voting data

# Load and plot the US Congress voting data
load("uscongress.RData")

cols <- c("#252525", "#737373", "#bdbdbd")
J <- ncol(X)

labs <- 1:J
tab <- t( table(class, X[,1]) )
FREQ <- prop.table(tab, 2)
for ( j in 2:J ) {
  tab <- t( table(class, X[,j]) )
  FREQ <- cbind( FREQ,  prop.table(tab, 2) )
}
par( mar = c(3,2,1,3), xpd = TRUE )
xp <- barplot( FREQ, xaxs = "r", space = rep(c(0.5,0.01), J),
               xaxt = "n", col = cols, border = "white", axes = FALSE)
axis(2, line = -0.5)
mtext(labs, side = 1, at = seq(1.505,39.66, by = 2.51), 
      las = 1, adj = 0.5, padj = 0.5)
mtext("Variables", side = 1, at = 39.66/2, line = 1.5)
mtext(rep(c("D", "R"), J), side = 3, at = xp, line = 0, cex = 0.6)
legend(40.5, 0.8, fill = cols, legend = c("n", "y", "u"), bty = "n" )

# run lca on all the variables
Gmax <- 7
ctrlLCA <- controlLCA(tol = 1e-07, nrep = 20)
lca <- fitLCA(X, G = 1:Gmax, ctrlLCA = ctrlLCA)

Variable selection

# ClustMMDD ---------------------
backExp <- backward.explorer(X, Kmax = Gmax, Criterion = "CteDim", 
                             project = "lca_uscongress")            # enumerate models
outDimJump <- dimJump.R("lca_uscongress_ExploredModels.txt", 
                        N = nrow(X), h = 1, header = TRUE)
cte1 <- outDimJump[[1]][1]
selCMMDD <- model.selection.R("lca_uscongress_ExploredModels.txt", 
                              cte = cte1, header = TRUE)

# BIC
rmBIC <- which( selCMMDD["BIC", 4:19] != 1 )
fitCMMDD_BIC <- fitLCA( X[,-rmBIC], G = selCMMDD["BIC", "K"], ctrlLCA = ctrlLCA )

# CteDim
rmCTE <- which( selCMMDD["CteDim", 4:19] != 1 )
fitCMMDD <- fitLCA( X[,-rmCTE], G = selCMMDD["CteDim", "K"], ctrlLCA = ctrlLCA )


# LCAvarsel ---------------------
# global independence assumption - Dean and Raftery (2010)
lcavslind <- LCAvarsel(X, G = 1:Gmax, independence = TRUE, 
                       ctrlLCA = ctrlLCA, parallel = TRUE)

# no global independence and swap-stepwise search - Fop, Smart, Murphy (2017)
lcavslswap <- LCAvarsel(X, G = 1:Gmax, swap = TRUE, 
                        ctrlLCA = ctrlLCA, parallel = TRUE)


# VarSelLCM ---------------------
fitlcm <- list()
for ( g in 1:Gmax ) {
  fitlcm[[g]] <- VarSelCluster(X, g, 
                               crit.varsel = "MICL", 
                               nbcores = 8, 
                               initModel = 200)  # increase it to avoid local optima
  print(g)
}
micl <- sapply(fitlcm, function(f) f@criteria@MICL)
best <- which.max(micl)

Results

# selected variables
cn <- colnames(X)
#
selCMMDD_BIC <- ifelse( is.na(match(cn, cn[-rmBIC])), 0, 1 )
selCMMDD <- ifelse( is.na(match(cn, cn[-rmCTE])), 0, 1 )
#
selvslind <- ifelse( is.na(match(cn, lcavslind$variables)), 0, 1 )
selvslswap <- ifelse( is.na(match(cn, lcavslswap$variables)), 0, 1 )
#
selmicl <- fitlcm[[best]]@model@omega     # VarSelLCM
#
SEL <- matrix( c(selCMMDD_BIC, selCMMDD, selvslind, selvslswap, selmicl), 
               byrow = TRUE, ncol = J )


# classification
ARI <- c(compareCluster(class, fitCMMDD$class)$adjRandIndex,
         compareCluster(class, fitCMMDD_BIC$class)$adjRandIndex,
         compareCluster(class, lcavslind$model$class)$adjRandIndex,
         compareCluster(class, lcavslswap$model$class)$adjRandIndex,
         compareCluster(class, fitlcm[[best]]@partitions@zMAP)$adjRandIndex
)


# number of clusters
G <- c(fitCMMDD_BIC$G, fitCMMDD$G, lcavslind$model$G, lcavslswap$model$G, best)


# plot
labs <- 1:J
met <- c("ClustMMDD-bic", "ClustMMDD", "LCAvarsel-ind", "LCAvarsel", "VarSelLCM")
k <- length(met)

par( mar = c(3,6.5,1,6) )
image( 1:J, 1:k, t(SEL), col = c("white", "#252525"), axes = FALSE, xlab = "", ylab = "" )
abline(v = 1:J + 0.5, col = "white", lwd = 2)
abline(h = 1:(k-1) + 0.5, col = "white", lwd = 4)
abline(h = c(0,k) + 0.5, col = "black", lwd = 2)
abline(v = c(0,J) + 0.5, col = "black", lwd = 2)
mtext(labs, side = 1, at = 1:J, las = 1)
mtext(met, side = 2, at = 1:k, las = 1, line = 0.4)
mtext(paste0("G = ", G), side = 4, at = 1:k + 0.1, las = 1, line = 1, cex = 0.8)
mtext(paste0("ARI = ", round(ARI,2)), side = 4, at = 1:k -0.2, las = 1, line = 1, cex = 0.8)
mtext("Variables", side = 1, at = J/2, line = 1.5, cex = 1)
